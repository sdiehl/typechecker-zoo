# Base literals
5
true
1 + 2

# Variables and let
let x = 5 in x + 1
let x = 10 in x * x

# Lambdas with refined argument types
\x : { n : Int | n > 0 } -> x
\x : Int -> x + 1

# Dependent application with positive argument
(\x : { n : Int | n > 0 } -> x) 3

# Built-in division by non-zero literal
(\x : { n : Int | n != 0 } -> div 10 x) 2

# Path conditions from if
\x : Int -> if x > 0 then x else 0 - x

# Negative: divide by zero
div 10 0

# Negative: pass possibly-zero where non-zero required
\x : Int -> div 10 x

# Negative: subtype mismatch on argument
let y = 0 in (\x : { n : Int | n > 0 } -> x) y

# Negative: pass negative where positive required
(\x : { n : Int | n > 0 } -> x) (0 - 5)
